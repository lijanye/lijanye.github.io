# 面试题56 - I. 数组中数字出现的次数

题目地址


/**
 * @param {number[]} nums
 * @return {number[]}
 */
var singleNumbers = function(nums) {
    let obj = {};
    let res = [];

    for(let i=0; i<nums.length; i++) {
        let item = nums[i];

        obj[item] = obj[item] ? obj[item] + 1 : 1;
    }

    for(const key in obj) {
        if(res.length === 2) break;
        if(obj[key] === 1) {
            res.push(key);
        }
    }

    return res;
};
