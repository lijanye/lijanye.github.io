# 86. 分隔链表

题目地址


/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} x
 * @return {ListNode}
 */
var partition = function(head, x) {

    if(!head) return null;
    let left = [];
    let right = [];

    let node = head;

    while(node) {
        if(node.val < x) {
            left.push(new ListNode(node.val));
        }else {
            right.push(new ListNode(node.val));
        }
        node = node.next;
    }

    let res = left.concat(right);

    for(let i=0; i<res.length; i++) {
        res[i].next = res[i+1];
    }

    return res[0];
};
